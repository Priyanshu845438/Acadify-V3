// --- Counter Animation ---
const counter = document.getElementById('satisfaction-counter');
function animateCounter() {
  let currentValue = 0;
  const targetValue = 98;
  const interval = setInterval(() => {
    if (currentValue >= targetValue) {
      clearInterval(interval);
    } else {
      currentValue++;
      counter.textContent = currentValue + '%';
    }
  }, 20); // Adjust speed here
}

// Use Intersection Observer to trigger counter only when visible
const observer = new IntersectionObserver((entries) => {
  entries.forEach(entry => {
    if (entry.isIntersecting) {
      animateCounter();
      observer.unobserve(entry.target); // Animate only once
    }
  });
}, { threshold: 0.5 });
observer.observe(counter);


// --- Mouse Parallax Effect ---
const heroSection = document.querySelector('.hero-section');
const videoWrapper = document.querySelector('.hero-video-wrapper');
heroSection.addEventListener('mousemove', function(e) {
  if (window.innerWidth > 992) { // Only apply on desktop
    const { clientX, clientY } = e;
    const { offsetWidth, offsetHeight } = heroSection;
    
    // Calculate movement amount (adjust divisor for more/less effect)
    const xMovement = (clientX / offsetWidth - 0.5) * 20; 
    const yMovement = (clientY / offsetHeight - 0.5) * 20;

    videoWrapper.style.transform = `translate(${xMovement}px, ${yMovement}px)`;
  }
});
4. Mobile Responsive Instructions

Video Performance: Autoplaying video can be heavy on mobile data and performance. The provided HTML already includes a best practice:

The <video> is hidden on mobile (d-none d-lg-block).

A static <img> fallback is shown instead (d-lg-none). This gives a great visual without the performance cost.

Layout Stacking: The Bootstrap grid (col-lg-6) will automatically stack the columns on mobile. The text content will appear on top of the image.

Text Alignment: The text-center text-lg-start classes center the text and CTAs on mobile/tablet for a balanced look, then switch back to left-aligned on desktop.

Parallax Effect: The JavaScript is written to only apply the parallax effect on screens wider than 992px, so it won't run on mobile where there is no mouse to track.
